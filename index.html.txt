        <div id="gemini-box" class="space-y-4">
            <div id="chat" class="h-64 overflow-y-auto glass p-4 rounded-xl text-sm space-y-2"></div>
            <div class="flex gap-2">
                <input id="g-input" class="flex-1 bg-slate-900 border border-white/10 p-3 rounded-xl outline-none" placeholder="Ask Gemini...">
                <button onclick="askGemini()" class="bg-indigo-600 px-6 rounded-xl font-bold">Send</button>
            </div>
        </div>

        <div id="zyphra-box" class="hidden space-y-4">
            <textarea id="z-input" class="w-full h-32 bg-slate-900 border border-white/10 p-4 rounded-xl outline-none" placeholder="Text to speech..."></textarea>
            <button onclick="speakZyphra()" class="w-full bg-purple-600 py-4 rounded-xl font-bold">Generate Audio</button>
        </div>

    <script>
        const GEMINI_KEY = "YOUR_GEMINI_KEY";
        const ZYPHRA_KEY = "YOUR_ZYPHRA_KEY";

        function tab(t) {
            document.getElementById('gemini-box').classList.toggle('hidden', t !== 'gemini');
            document.getElementById('zyphra-box').classList.toggle('hidden', t !== 'zyphra');
        }

        async function askGemini() {
            const inp = document.getElementById('g-input');
            const chat = document.getElementById('chat');
            if(!inp.value) return;

            document.getElementById('visualizer').classList.add('animating');
            chat.innerHTML += `<div><b>You:</b> ${inp.value}</div>`;
            
            try {
                const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${GEMINI_KEY}`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({contents: [{parts: [{text: inp.value}]}]})
                });
                const data = await res.json();
                chat.innerHTML += `<div class="text-indigo-400"><b>AI:</b> ${data.candidates[0].content.parts[0].text}</div>`;
            } catch (e) { chat.innerHTML += `<div class="text-red-500">Error! Check API Key.</div>`; }
            
            document.getElementById('visualizer').classList.remove('animating');
            inp.value = '';
            chat.scrollTop = chat.scrollHeight;
        }

        async function speakZyphra() {
            const text = document.getElementById('z-input').value;
            if(!text) return;
            document.getElementById('visualizer').classList.add('animating');
            // Logic for Zyphra call goes here (similar to previous steps)
            alert("Vercel par CORS ka masla proxy ke bina hal ho sakta hai!");
            document.getElementById('visualizer').classList.remove('animating');
        }
    </script>
